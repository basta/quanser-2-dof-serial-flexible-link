% ======================================================================
%  Quanser 2-DOF Serial Flexible Link  --  minimal schematic.
%  Top view (horizontal plane).  Serial elbow arm:
%    base -[motor 1]- flexible link 1 -[motor 2 = elbow]- flexible link 2
%    - tip mass.
%  Each beam = rigid link on a torsional root spring (Ks_i); an encoder
%  reads the joint/hub angle theta_h, a strain gauge the deflection theta_t.
%
%  Compile:  pdflatex model_diagram.tex
% ======================================================================
\documentclass[border=4pt]{standalone}
\usepackage{amsmath}
\usepackage{xcolor}
\usepackage{tikz}
\usetikzlibrary{arrows.meta,calc}

\definecolor{hubcol}{HTML}{1f77b4}
\definecolor{linkcol}{HTML}{2ca02c}
\definecolor{j2col}{HTML}{7b3294}
\definecolor{link2col}{HTML}{117777}
\definecolor{tipcol}{HTML}{d62728}
\definecolor{motcol}{HTML}{555555}
\definecolor{refcol}{HTML}{999999}
\definecolor{springcol}{HTML}{ff7f0e}

\begin{document}
\begin{tikzpicture}[
    >={Stealth[length=2.4mm,width=2mm]},
    font=\small,
  ]

  % ---- geometry (deflections exaggerated for clarity) ----------------
  \def\hA{14}\def\aA{22}\def\lA{3.7}   % joint 1: hub / link angle, length
  \def\hBrel{14}\def\aBrel{22}\def\lB{3.0} % joint 2 (rel. to link 1)

  \coordinate (O1) at (0,0);
  \coordinate (E)  at (\aA:\lA);

  % ---- variable key (top-left corner) --------------------------------
  \node[anchor=north west,align=left,font=\scriptsize,inner xsep=1pt,
        inner ysep=2pt]
        at (-1.95,4.35)
        {\textcolor{motcol}{$u_1$}, \textcolor{j2col}{$u_2$} — motor currents (inputs)\\
         \textcolor{hubcol}{$\theta_{h1}$}, \textcolor{j2col}{$\theta_{h2}$} — joint angles (encoders)\\
         \textcolor{tipcol}{$\theta_{t1}$}, \textcolor{tipcol}{$\theta_{t2}$} — link deflections (strain gauges)};

  % ---- fixed reference (joint-1 zero) --------------------------------
  \draw[refcol,dashed,->,line width=0.6pt] (0,0) -- (4.6,0)
       node[right,refcol] {$0^\circ$};

  % ---- fixed base / ground -------------------------------------------
  \begin{scope}
    \clip (-0.45,-0.5) rectangle (0.45,-0.16);
    \foreach \x in {-0.5,-0.35,...,0.5}
      \draw[motcol,line width=0.4pt] (\x,-0.16) -- (\x-0.2,-0.38);
  \end{scope}
  \draw[motcol,line width=0.8pt] (-0.45,-0.16) -- (0.45,-0.16);

  % ====================================================================
  %  JOINT 1
  % ====================================================================
  \node[draw=motcol,fill=motcol!8,rounded corners=1.5pt,inner sep=2.5pt,
        font=\scriptsize\sffamily,anchor=north] (m1) at (0,-0.55) {motor 1};
  \draw[motcol,->,line width=1.0pt] (-1.9,-1.35) -- (m1.south west);
  \node[motcol,anchor=east,font=\small] at (-1.9,-1.35) {$u_1$};

  \draw[hubcol,dash pattern=on 4pt off 2pt,line width=0.7pt]
        (O1) -- (\hA:\lA-0.3);
  \fill[hubcol] (O1) circle (1.4pt);

  \draw[linkcol,line width=2.2pt,line cap=round] (O1) -- (E);
  \node[linkcol,font=\scriptsize,anchor=south,rotate=\aA]
        at (\aA:\lA*0.5) {link 1};

  \draw[hubcol,line width=0.8pt] (1.85,0) arc (0:\hA:1.85);
  \node[hubcol] at (\hA*0.5:2.1) {$\theta_{h1}$};
  \draw[tipcol,line width=0.8pt] (\hA:2.65) arc (\hA:\aA:2.65);
  \node[tipcol] at ({(\hA+\aA)/2}:2.95) {$\theta_{t1}$};

  % ====================================================================
  %  JOINT 2 + LINK 2  (frame aligned with link 1)
  % ====================================================================
  \begin{scope}[shift={(E)},rotate=\aA]
    \coordinate (T) at (\aBrel:\lB);
    \fill[j2col] (0,0) circle (1.4pt);

    \node[draw=j2col,fill=j2col!8,rounded corners=1.5pt,inner sep=2.5pt,
          font=\scriptsize\sffamily,anchor=north,rotate=-\aA]
          (m2) at (0,-0.46) {motor 2};
    \draw[j2col,line width=0.9pt] (0,0) -- (m2.north);   % motor 2 sits at the elbow

    % link-1 direction continued past the elbow (reference for theta_h2)
    \draw[linkcol!65,dash pattern=on 3pt off 2pt,line width=0.6pt]
          (0,0) -- (0:\lB-0.5);
    % hub-2 commanded direction
    \draw[j2col,dash pattern=on 4pt off 2pt,line width=0.7pt]
          (0,0) -- (\hBrel:\lB-0.4);

    \draw[link2col,line width=2.0pt,line cap=round] (0,0) -- (T);
    \pgfmathsetmacro\aBabs{\aA+\aBrel}   % link-2 absolute angle
    \node[link2col,font=\scriptsize,anchor=south,rotate=\aBabs]
          at (\aBrel:\lB*0.5) {link 2};

    \draw[j2col,line width=0.7pt] (1.45,0) arc (0:\hBrel:1.45);
    \node[j2col,font=\scriptsize] at (\hBrel*0.5:1.7) {$\theta_{h2}$};
    \draw[tipcol,line width=0.7pt] (\hBrel:2.05) arc (\hBrel:\aBrel:2.05);
    \node[tipcol,font=\scriptsize] at ({(\hBrel+\aBrel)/2}:2.3) {$\theta_{t2}$};

    \begin{scope}[shift={(T)},rotate=\aBrel]
      \draw[motcol,fill=motcol!20,rounded corners=1pt]
            (-0.17,-0.2) rectangle (0.17,0.2);
    \end{scope}
    \node[motcol,font=\scriptsize,anchor=south east,rotate=-\aA]
          at ($(T)+(\aBrel+90:0.3)$) {tip mass};
  \end{scope}

  % ---- input current to motor 2 (points into the motor) --------------
  \draw[j2col,->,line width=1.0pt] ($(m2.east)+(0.85,-0.35)$) -- (m2.east);
  \node[j2col,anchor=west,font=\small] at ($(m2.east)+(0.9,-0.37)$) {$u_2$};

\end{tikzpicture}
\end{document}
